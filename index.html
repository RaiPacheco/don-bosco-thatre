// --- CONFIGURATION --- const ROWS = ["A", "B", "C", "D", "E"]; const SEATS_PER_ROW = 8; const PRICES = { adult: 15, minor: 8 }; const TAKEN_SEATS = ["A1", "B3", "C5"]; let selectedSeats = {}; // Format: { "A2": "adult" } // --- DOM ELEMENTS --- const grid = document.getElementById("seatGrid"); const selectedContainer = document.getElementById("selectedSeats"); const totalPriceDisplay = document.getElementById("totalPrice"); const confirmBtn = document.getElementById("confirmBtn"); // --- INITIALIZE SEATS --- function initGrid() { ROWS.forEach(row => { for (let i = 1; i <= SEATS_PER_ROW; i++) { const seatId = row + i; const isTaken = TAKEN_SEATS.includes(seatId); const seat = document.createElement("div"); seat.innerText = seatId; seat.className = `seat text-center py-2 rounded-lg text-sm font-medium ${isTaken ? "taken cursor-not-allowed" : "bg-gray-800 hover:bg-gray-700 cursor-pointer"}`; if (!isTaken) { seat.onclick = () => toggleSeat(seatId, seat); } grid.appendChild(seat); } }); } // --- LOGIC FUNCTIONS --- function toggleSeat(seatId, element) { if (selectedSeats[seatId]) { delete selectedSeats[seatId]; element.classList.remove("selected"); } else { selectedSeats[seatId] = "adult"; element.classList.add("selected"); } renderSelectedSeats(); } function changeType(seatId, newType) { selectedSeats[seatId] = newType; renderSelectedSeats(); } function renderSelectedSeats() { selectedContainer.innerHTML = ""; let total = 0; Object.keys(selectedSeats).forEach(seatId => { const type = selectedSeats[seatId]; total += PRICES[type]; const row = document.createElement("div"); row.className = "flex justify-between items-center"; row.innerHTML = ` ${seatId} `; selectedContainer.appendChild(row); }); totalPriceDisplay.innerText = total; } // --- EVENT HANDLERS --- confirmBtn.onclick = () => { const seatCount = Object.keys(selectedSeats).length; if (seatCount === 0) { alert("Please select at least one seat."); return; } console.log("Booking Confirmed:", selectedSeats); alert(`Booking confirmed for ${seatCount} seat(s)! ✨`); }; // Start the app initGrid();
